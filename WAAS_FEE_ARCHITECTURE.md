### Pattern 2: Simple Modal (MakeOfferModal, CreateListingModal)

```typescript
function MakeOfferModal() {
  const waasFee = useWaasFeeManagement({
    chainId,
    enabled: isWaaS,
  });
  
  // Show fee UI if manual confirmation needed
  if (waasFee.needsManualConfirmation) {
    return (
      <div>
        <h2>Select Fee Token</h2>
        <SelectWaasFeeOptions
          chainId={chainId}
          feeOptionConfirmation={waasFee.feeOptionConfirmation!}
          selectedOption={waasFee.selectedOption}
          onSelectedOptionChange={waasFee.setSelectedOption}
          onConfirm={() => {
            waasFee.confirmFeeOption(
              waasFee.feeOptionConfirmation!.id,
              waasFee.selectedOption!.token.contractAddress
            );
            waasFee.setOptionConfirmed(true);
          }}
          optionConfirmed={waasFee.optionConfirmed}
        />
      </div>
    );
  }
  
  // Show main modal UI
  return (
    <div>
      <MakeOfferForm />
      <Button
        onClick={handleSubmit}
        disabled={!waasFee.isReady}
      >
        Make Offer
      </Button>
    </div>
  );
}
```

### Pattern 3: Transaction Integration

```typescript
// In any modal's transaction handler
function useHandleTransaction() {
  const waasFee = useWaasFeeManagement({
    chainId,
    enabled: isWaaS,
  });
  
  const processStep = useProcessStep({
    // ... other config
    waasFeeConfirmation: waasFee.getConfirmationState(),
  });
  
  const handleExecute = async () => {
    // Check if ready
    if (!waasFee.isReady) {
      console.warn('Fee not confirmed yet');
      return;
    }
    
    // Execute with fee confirmation state
    await processStep.execute();
  };
  
  return { handleExecute };
}
```
